    Core → View → ListView

    View for the collection of items.

    Example:

    class ListItemView extends View {
  html(data) {
    return `<li>${data.title}</li>`;
  }
}

let instanceView = new ListView(Node, ListItemView,
          {
            data: [
              { title: 'foo' },
              { ttile: 'bar' }
            ]
          });
